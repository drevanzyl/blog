leafMap <- leaflet(results) %>% addTiles() %>%
  addCircleMarkers(
    lng=~longitude, lat=~latitude,
    color = ~pal(insp_type),
    stroke = FALSE, fillOpacity = 0.5,
    clusterOptions = markerClusterOptions()
